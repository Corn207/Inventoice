%\makeglossaries
\makenoidxglossaries

% Danh mục thuật ngữ và từ viết tắt
\newglossaryentry{mvvm}{
    type=\acronymtype,
    name={MVVM},
    description={Mô hình thiết kế ứng dụng người dùng cuối (Model-View-ViewModel)},
}
\newglossaryentry{api}{
    type=\acronymtype,
    name={API},
    description={Giao diện lập trình ứng dụng (Application Programming Interface)}
}
\newglossaryentry{server}{
    type=\acronymtype,
    name={Server},
    description={Máy chủ trong mô hình Client-Server, thường ám chỉ máy tính có cấu hình mạnh, có nhiệm vụ xử lý các yêu cầu từ phía Client},
}
\newglossaryentry{client}{
    type=\acronymtype,
    name={Client},
    description={Máy khách trong mô hình Client-Server, thường gửi các yêu cầu đến Server và nhận kết quả từ Server},
}
\newglossaryentry{framework}{
    type=\acronymtype,
    name={Framework},
    description={Bộ khung phát triển ứng dụng, là một tập các thư viện, công cụ, hàm, lớp, ... giúp cho việc phát triển ứng dụng nhanh hơn}
}
\newglossaryentry{nosql}{
    type=\acronymtype,
    name={NoSQL},
    description={Hệ quản trị cơ sở dữ liệu phi quan hệ (Not Only SQL)}
}
\newglossaryentry{backend}{
    type=\acronymtype,
    name={Backend},
    description={Phía máy chủ, thường là phần xử lý logic, xử lý dữ liệu, ...}
}
\newglossaryentry{frontend}{
    type=\acronymtype,
    name={Frontend},
    description={Phía máy khách, thường là phần giao diện người dùng}
}
\newglossaryentry{oop}{
    type=\acronymtype,
    name={OOP},
    description={Lập trình hướng đối tượng (Object Oriented Programming)}
}
